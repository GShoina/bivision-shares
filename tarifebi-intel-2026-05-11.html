Scraper Pipeline — PHP Pseudo-code (reconstructed) 
// cron: */4 * * * * php /srv/tarifebi/scraper/run.php

class StoreScraper {

  // Carrefour — internal JSON API (session cookie)
  public function fetchCarrefour(): array {
    return json_decode($this->httpGet(
      'https://www.carrefour.ge/api/2.0/catalog/product/list/',
      ['lang'=>'ka','pageSize'=>100,'currentPage'=>$page],
      ['Cookie' => $this->getSession('carrefour')]
    ), true)['items'];
  }

  // Aversi — Cloudflare bypass via Playwright headless
  public function fetchAversi(): array {
    $browser = $this->playwright->launch(['headless' => true]);
    $context = $browser->newContext(['storageState' => 'sessions/aversi.json']);
    $page    = $context->newPage();
    $page->goto('https://aversi.ge/products?page=' . $page_num);
    $page->waitForSelector('.product-card');
    $html = $page->content();
    $context->storageState('sessions/aversi.json'); // persist cookies
    $browser->close();
    return $this->parseHtml($html);
  }

  // Europroduct — plain HTML, no API
  public function fetchEuroproduct(): array {
    $html = $this->httpGet('https://europroduct.ge/products?page=' . $page_num);
    return $this->parseHtml($html); // DOMDocument
  }

  // Image normalization → MD5 → S3
  private function normalizeImage(string $src): string {
    $data  = file_get_contents($src);
    $key   = "products/" . md5($data) . "." . $this->guessExt($src);
    if (!$this->s3->exists($key)) $this->s3->put($key, $data, 'public-read');
    return "https://s3.iserv.ge/svc-tarifebi/{$key}";
  }

  // EAN barcode extract from name
  private function extractBarcode(string $name): ?string {
    return preg_match('/\/(\d{8,14})$/', $name, $m) ? $m[1] : null;
  }

  // Upsert to DB
  private function upsertProduct(array $raw, int $storeId, int $cityId): void {
    $this->db->query("
      INSERT INTO products (name,barcode,image_url,original_price,sale_price,
        discount_pct,is_on_sale,store_id,category_id,city_id,scraped_at)
      VALUES (?,?,?,?,?,?,?,?,?,?,NOW())
      ON DUPLICATE KEY UPDATE
        sale_price=VALUES(sale_price), original_price=VALUES(original_price),
        discount_pct=VALUES(discount_pct), scraped_at=NOW()
    ", [$raw['name'], $this->extractBarcode($raw['name']),
        $this->normalizeImage($raw['image_url']),
        $raw['original_price'], $raw['sale_price'],
        $raw['discount_pct'], $raw['is_on_sale'],
        $storeId, $raw['category_id'], $cityId]);
  }
}

# scraper architecture — Python + Playwright + rotating proxy
class BiScraper:

    # Tier 1: public JSON APIs (no browser needed)
    TIER1_TARGETS = {
        'carrefour': 'https://www.carrefour.ge/api/products',
        'spar':      'https://spar.ge/api/v2/catalog',
        'goodwill':  'https://goodwill.ge/api/items',
    }

    # Tier 2: CF-protected sites → Playwright persistent session
    TIER2_CF_TARGETS = ['aversi.ge', 'psp.ge']

    def scrape_tier2(self, url: str):
        with sync_playwright() as p:
            browser = p.chromium.launch_persistent_context(
                user_data_dir='/var/scraper/sessions',  # ← persist cookies
                proxy={'server': self.get_rotating_proxy()},
                args=['--disable-blink-features=AutomationControlled']
            )
            page = browser.new_page()
            page.goto(url)
            page.wait_for_load_state('networkidle')
            return page.content()

    # Image normalization — ALL assets to own S3
    def normalize_image(self, src_url: str) -> str:
        img_bytes = requests.get(src_url, timeout=10).content
        md5 = hashlib.md5(img_bytes).hexdigest()
        key = f"products/{md5}.webp"
        # convert to WebP + upload
        img = Image.open(BytesIO(img_bytes)).convert('RGB')
        buf = BytesIO()
        img.save(buf, 'WEBP', quality=82)
        self.s3.put_object(Key=key, Body=buf.getvalue(), ContentType='image/webp')
        return f"https://cdn.bivision.ge/{key}"

    # EAN extraction — from name field
    def extract_ean(self, name: str) -> str | None:
        match = re.search(r'/(\d{8,13})$', name)
        return match.group(1) if match else None

def get_image_url(key: str, client_id: str) -> str:

        'get_object',
        Params={
            'Bucket': 'bivision-data',
            'Key': key,

CANARY_PRODUCTS = [
    # fake products — never scraped from real stores
    # each client sees slightly different version
    {"id": 999001, "name": "Bivision Monitor Milk 1L", "price": 99999.99, "store": "carrefour"},
    {"id": 999002, "name": "Bivision Sentinel Water 500ml", "price": 88888.88, "store": "spar"},
]

def inject_canary(response: dict, client_id: str) -> dict:
    """Inject unique canary per client into response"""
    canary = CANARY_PRODUCTS[hash(client_id) % len(CANARY_PRODUCTS)].copy()
    # Tweak price slightly per client — unique fingerprint
    canary['price'] += hash(client_id + 'salt') % 100 / 1000
    canary['client_tag'] = client_id  # invisible in normal view
    response['items'].append(canary)
    return response

def check_client_behavior(client_id: str, window_minutes: int = 60):
