// cron: */4 * * * * php /var/www/tarifebi/scraper/run.php
// runs every 4 hours, one store at a time

class StoreScraper {

    // ── STEP 1: Fetch products from source ──────────────────────────────
    public function fetchCarrefour(): array {
        // Carrefour.ge has internal JSON API (auth via session cookie)
        // rotates User-Agent, maintains session
        $response = $this->httpGet('https://www.carrefour.ge/api/2.0/catalog/product/list/', [
            'lang'        => 'ka',
            'pageSize'    => 100,
            'currentPage' => $page,
        ], $headers = ['Cookie' => $this->getSession('carrefour')]);

        return json_decode($response, true)['items'];
    }

    public function fetchAversi(): array {
        // aversi.ge = Cloudflare managed challenge → needs real browser
        // Playwright headless, Georgian IP, persistent context (cookie reuse)
        $browser = $this->playwright->launch(['headless' => true]);
        $context = $browser->newContext(['storageState' => 'sessions/aversi.json']);
        $page    = $context->newPage();
        $page->goto('https://aversi.ge/products?page=' . $page_num);
        $page->waitForSelector('.product-card');
        $html = $page->content();
        $context->storageState('sessions/aversi.json'); // save cookies
        $browser->close();
        return $this->parseHtml($html);
    }

    public function fetchEuroproduct(): array {
        // europroduct.ge = plain HTML, no API, Bootstrap/jQuery site
        $html = $this->httpGet('https://europroduct.ge/products?page=' . $page_num);
        return $this->parseHtml($html);  // DOMDocument / simple_html_dom
    }

    // ── STEP 2: Normalize image ──────────────────────────────────────────
    private function normalizeImage(string $sourceUrl): string {
        $imageData = file_get_contents($sourceUrl);
        $md5       = md5($imageData);
        $ext       = $this->guessExt($sourceUrl);  // jpg / webp / png
        $s3Key     = "products/{$md5}.{$ext}";

        if (!$this->s3->exists($s3Key)) {
            $this->s3->put($s3Key, $imageData, 'public-read');
        }
        return "https://s3.iserv.ge/svc-tarifebi/{$s3Key}";
    }

    // ── STEP 3: Extract barcode from name ────────────────────────────────
    private function extractBarcode(string $name): ?string {
        // pattern: "პროდუქტი სახელი/4860001002023"
        if (preg_match('/\/(\d{8,14})$/', $name, $m)) {
            return $m[1];  // EAN-13 or EAN-8
        }
        return null;
    }

    // ── STEP 4: Upsert to DB ─────────────────────────────────────────────
    private function upsertProduct(array $raw, int $storeId, int $cityId): void {
        $barcode  = $this->extractBarcode($raw['name']);
        $imageUrl = $this->normalizeImage($raw['image_url']);

        $this->db->query("
            INSERT INTO products
              (name, barcode, image_url, original_price, sale_price,
               discount_pct, is_on_sale, store_id, category_id, city_id, scraped_at)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, NOW())
            ON DUPLICATE KEY UPDATE
              sale_price    = VALUES(sale_price),
              original_price= VALUES(original_price),
              discount_pct  = VALUES(discount_pct),
              is_on_sale    = VALUES(is_on_sale),
              scraped_at    = NOW()
        ", [$raw['name'], $barcode, $imageUrl, $raw['original_price'],
            $raw['sale_price'], $raw['discount_pct'], $raw['is_on_sale'],
            $storeId, $raw['category_id'], $cityId]);
    }
}
